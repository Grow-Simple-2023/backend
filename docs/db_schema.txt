user: 
	id: ObjectID, 
	name {
		first: str, 
		last: str
	}
	phone_no: str, 
	password: str, 
	role: enum[ADMIN, RIDER]

item: 
	id: ObjectID, 
	title: str, 
	description {
		length: float, 
		breadth: float, 
		height: float, 
		weight: float
	}, 
	address: str, 
	location: {
		latitude: float, 
		longitude: float
	}, 
	EDD: datetime, 
	control {
		is_fulfilled: bool, 
		is_delivery: bool, 
		is_pickup: bool, 
		is_assigned: bool, 
		is_cancelled: bool
	},
	delivered_on: datetime
	OTP: int [same OTP for same (address, EDD)]

route: 
	id: ObjectID, 
	rider_id ObjectID (from user_id), 
	rider_location: List[{
		latitude: float, 
		longitude: float,
		time: long long int
	}]
	bag_description {
		length: float, 
		breadth: float, 
		height: float
	}, 
	item_ids: List[ObjectID] (From Item id)
	route_end_otp: int

route_archive: ! Same as route but for routes that are completed !
